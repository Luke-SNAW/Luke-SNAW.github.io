그리고 스키마에서 이 템플릿을 연결:

version: 1
schemas:
  - id: daily
    parent: root
    children:
      - pattern: journal
        children:
          - pattern: "[0-2][0-9][0-9][0-9]"
            children:
              - pattern: "[0-1][0-9]"
                children:
                  - pattern: "[0-3][0-9]"
                    template: templates.daily
